NLP Mini App: Build a chatbot, fake news detector, or keyword extractor.


Chatbot Program:

def chatbot():
    print("Chatbot: Hello! Type 'bye' to exit.")
    
    while True:
        user_input = input("You: ").lower()
        
        if "hello" in user_input or "hi" in user_input:
            print("Chatbot: Hello! How can I help you?")
        
        elif "how are you" in user_input:
            print("Chatbot: I'm just a bot, but I'm doing great!")
        
        elif "your name" in user_input:
            print("Chatbot: I am a simple NLP chatbot.")
        
        elif "help" in user_input:
            print("Chatbot: I can answer basic questions like greetings.")
        
        elif "bye" in user_input:
            print("Chatbot: Goodbye! Have a nice day.")
            break
        
        else:
            print("Chatbot: Sorry, I didn't understand that.")

# Run chatbot
chatbot()
